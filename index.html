                d += ` C ${cp1x},${cp1y} ${cp2x},${cp2y} ${nextX},${nextY}`;

                currentX = nextX;

            }

            path.setAttribute('d', d);

        }

        let timerInterval;

        function startВыбросTimer(e) {

            e.stopPropagation(); // предотвращаем закрытие шага

            const btn = document.getElementById('timerBtn');

            const display = document.getElementById('timerVal');

            

            if (timerInterval) clearInterval(timerInterval);

            

            let timeLeft = 2.00;

            btn.disabled = true;

            btn.innerText = "идет выброс напряжения...";

            

            timerInterval = setInterval(() => {

                timeLeft -= 0.05;

                if (timeLeft <= 0) {

                    timeLeft = 0;

                    clearInterval(timerInterval);

                    btn.innerText = "выброс завершен";

                    display.style.color = "#c9a96e";

                }

                display.innerText = timeLeft.toFixed(2);

            }, 50);

        }
